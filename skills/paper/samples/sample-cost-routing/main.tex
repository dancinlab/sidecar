\section{Formula}
For $N$ tasks with per-task tier $t_i$ and Anthropic per-call rate $r(t_i)$, total cost is the trivial sum
\begin{equation}
  C(N) = \sum_{i=1}^{N} r(t_i) = \sum_{t \in \{\text{haiku, sonnet, opus}\}} N_t \, r(t).
\end{equation}
Savings of router $R$ versus baseline $B$:
\begin{equation}
  s_{R/B} = \frac{C_B - C_R}{C_B} \cdot 100\,\%.
\end{equation}

\section{Method}
\textbf{Task manifest:} N fixed prompts spanning trivial-fact, short-code, longer-reasoning. \textbf{Tier rule:} length-based cutoffs $\le L_1$ → haiku, $\le L_2$ → sonnet, else opus. \textbf{Persistence:} verbatim \texttt{claude --bare -p --output-format json} stdout under \texttt{.verdicts/<slug>/\{baseline,router\}.tsv}.
